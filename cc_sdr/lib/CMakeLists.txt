########################################################################
# Setup library
########################################################################
include(GrPlatform) #define LIB_SUFFIX

include_directories(/opt/ground_station/include)
include_directories(${Boost_INCLUDE_DIR})
link_directories(${Boost_LIBRARY_DIRS})
link_directories(/opt/ground_station/lib)
list(APPEND cc_sdr_sources
    cc_decoder_impl.cc
    rs_work.c
    convolutional_coder.c)

set(cc_sdr_sources "${cc_sdr_sources}" PARENT_SCOPE)
if(NOT cc_sdr_sources)
	MESSAGE(STATUS "No C++ sources... skipping lib/")
	return()
endif(NOT cc_sdr_sources)

add_library(gnuradio-cc_sdr SHARED ${cc_sdr_sources})
target_link_libraries(gnuradio-cc_sdr ${Boost_LIBRARIES} ${GNURADIO_ALL_LIBRARIES})
set_target_properties(gnuradio-cc_sdr PROPERTIES DEFINE_SYMBOL "gnuradio_cc_sdr_EXPORTS")

if(APPLE)
    set_target_properties(gnuradio-cc_sdr PROPERTIES
        INSTALL_NAME_DIR "${CMAKE_INSTALL_PREFIX}/lib"
    )
endif(APPLE)

########################################################################
# Install built library files
########################################################################
install(TARGETS gnuradio-cc_sdr
    LIBRARY DESTINATION lib${LIB_SUFFIX} # .so/.dylib file
    ARCHIVE DESTINATION lib${LIB_SUFFIX} # .lib file
    RUNTIME DESTINATION bin              # .dll file
)

########################################################################
# Build and register unit test
########################################################################
include(GrTest)

include_directories(${CPPUNIT_INCLUDE_DIRS})

list(APPEND test_cc_sdr_sources
    ${CMAKE_CURRENT_SOURCE_DIR}/test_cc_sdr.cc
    ${CMAKE_CURRENT_SOURCE_DIR}/qa_cc_sdr.cc
)

add_executable(test-cc_sdr ${test_cc_sdr_sources})

target_link_libraries(
  test-cc_sdr
  ${GNURADIO_RUNTIME_LIBRARIES}
  ${Boost_LIBRARIES}
  ${CPPUNIT_LIBRARIES}
  gnuradio-cc_sdr
)

GR_ADD_TEST(test_cc_sdr test-cc_sdr)
